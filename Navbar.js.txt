export function Navbar({ cartCount, onCartClick }) {
const nav = document.createElement('nav');
nav.className = 'navbar glass';
nav.innerHTML = `
<div class="nav-content">
<div class="logo">PrimeMarket</div>
<button class="cart-btn" aria-label="Cart">
?? <span id="cart-count" class="badge">${cartCount}</span>
</button>
</div>
`;
nav.querySelector('.cart-btn').addEventListener('click', onCartClick);
return nav;
}
